title: "Quickstart"
description: "Get a working zoom element in under 5 minutes"
---

<Steps>
  <Step title="Install Zoomist">
    Install the package from npm:

    <CodeGroup>

    ```bash npm
    npm install zoomist
    ```

    ```bash yarn
    yarn add zoomist
    ```

    ```bash pnpm
    pnpm add zoomist
    ```

    </CodeGroup>
  </Step>

  <Step title="Add the HTML structure">
    Zoomist requires three nested elements. Add the following markup to your HTML:

    ```html
    <!-- zoomist-container -->
    <div class="zoomist-container">
      <!-- zoomist-wrapper is required -->
      <div class="zoomist-wrapper">
        <!-- zoomist-image is required -->
        <div class="zoomist-image">
          <!-- add any content you want to zoom here -->
          <img src="your-image.jpg" />
        </div>
      </div>
    </div>
    ```

    <Note>
      `zoomist-wrapper` and `zoomist-image` are required class names. Zoomist locates these elements by class to apply transforms.
    </Note>
  </Step>

  <Step title="Add CSS styles">
    Give the container and image explicit dimensions. Zoomist does not impose any default sizing:

    ```css
    .zoomist-container {
      width: 100%;
      max-width: 600px;
    }

    .zoomist-image {
      width: 100%;
      aspect-ratio: 1;
    }

    .zoomist-image img {
      width: 100%;
      height: 100%;
      object-fit: cover;
      object-position: center;
    }
    ```
  </Step>

  <Step title="Initialize Zoomist">
    Import the CSS, import the library, and create a new instance pointing at the container element:

    ```javascript
    import 'zoomist/css'
    import Zoomist from 'zoomist'

    const zoomist = new Zoomist('.zoomist-container', {
      // Optional parameters
      maxScale: 4,
      bounds: true,
      // Enable the slider module
      slider: true,
      // Enable the zoomer module
      zoomer: true
    })
    ```
  </Step>
</Steps>

## Complete example

Here is a self-contained example combining all three parts:

<CodeGroup>

```html index.html
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Zoomist example</title>
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/zoomist@2/zoomist.css" />
    <style>
      .zoomist-container {
        width: 100%;
        max-width: 600px;
      }

      .zoomist-image {
        width: 100%;
        aspect-ratio: 1;
      }

      .zoomist-image img {
        width: 100%;
        height: 100%;
        object-fit: cover;
        object-position: center;
      }
    </style>
  </head>
  <body>
    <div class="zoomist-container">
      <div class="zoomist-wrapper">
        <div class="zoomist-image">
          <img src="your-image.jpg" alt="Zoomable image" />
        </div>
      </div>
    </div>

    <script src="https://cdn.jsdelivr.net/npm/zoomist@2/zoomist.umd.js"></script>
    <script>
      const zoomist = new Zoomist('.zoomist-container', {
        maxScale: 4,
        bounds: true
      })
    </script>
  </body>
</html>
```

```javascript main.js (npm)
import 'zoomist/css'
import Zoomist from 'zoomist'

const zoomist = new Zoomist('.zoomist-container', {
  maxScale: 4,
  bounds: true
})
```

</CodeGroup>
